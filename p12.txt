f = function(x,y) x+y
x0 = 0
y0 = 1
h = 0.1
x_end = 0.2
x = seq(x0, x_end, by = h)
n = length(x)
y = numeric(n)
y[1] = y0
for(i in 1:(n-1))
{
  k1 = h * f(x[i], y[i])
  k2 = h * f(x[i] + h/2, y[i] + k1/2)
  k3 = h * f(x[i] + h/2, y[i] + k2/2)
  k4 = h * f(x[i] + h, y[i] + k3)
  y[i+1] = y[i] + (k1 + 2*k2 + 2*k3 + k4)/6
}
for(i in 1:n)
{
  cat("x =", x[i], " y =", y[i], "\n")
}
